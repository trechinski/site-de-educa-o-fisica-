// Mapeamento de elementos e lógica de registro com validações básicas const form = { name: () => document.getElementById("name"), email: () => document.getElementById("email"), password: () => document.getElementById("password"), confirmPassword: () => document.getElementById("confirm-password"), registerButton: () => document.getElementById("register-button"), backToLoginButton: () => document.getElementById("back-to-login"), emailInvalidError: () => document.getElementById("email-invalid-error"), passwordRequiredError: () => document.getElementById("password-required-error"), passwordLengthError: () => document.getElementById("password-length-error"), passwordMatchError: () => document.getElementById("password-match-error"), message: () => document.getElementById("message") }; function validateEmail(email) { return /\S+@\S+\.\S+/.test(email); } function showMessage(text, show = true) { const el = form.message(); if (!el) return; el.textContent = text; el.classList.toggle("visually-hidden", !show); } function toggleErrors() { const email = form.email().value.trim(); const password = form.password().value; const confirm = form.confirmPassword().value; form.emailInvalidError().classList.toggle("visually-hidden", validateEmail(email)); form.passwordRequiredError().classList.toggle("visually-hidden", !!password); form.passwordLengthError().classList.toggle("visually-hidden", password.length >= 6); form.passwordMatchError().classList.toggle("visually-hidden", password === confirm); } function isFormValid() { const email = form.email().value.trim(); const password = form.password().value; const confirm = form.confirmPassword().value; return validateEmail(email) && password && password.length >= 6 && password === confirm; } function mapFirebaseError(error) { switch (error.code) { case "auth/email-already-in-use": return "Este email já está em uso."; case "auth/invalid-email": return "Formato de email inválido."; case "auth/weak-password": return "Senha muito fraca."; default: return error.message || "Erro ao registrar usuário."; } } function registerUser() { toggleErrors(); if (!isFormValid()) { showMessage("Corrija os erros do formulário antes de continuar.", true); return; } const email = form.email().value.trim(); const password = form.password().value; // Se estiver usando Firebase, certifique-se de que o SDK foi inicializado antes if (window.firebase && firebase.auth) { firebase.auth().createUserWithEmailAndPassword(email, password) .then(userCredential => { showMessage("Registro concluído com sucesso. Redirecionando...", true); // redireciona ao login (caminho relativo correto a partir de src) setTimeout(() => window.location.href = "../../login.html", 1200); }) .catch(err => { showMessage(mapFirebaseError(err), true); }); } else { // Fallback local (desenvolvimento) console.warn("Firebase não encontrado — registro em modo mock."); showMessage("Registro simulado. (Firebase não inicializado)", true); } } document.addEventListener("DOMContentLoaded", () => { const btn = form.registerButton(); if (btn) btn.addEventListener("click", registerUser); const back = form.backToLoginButton(); if (back) back.addEventListener("click", () => window.location.href = "../../login.html"); // atualizar erros em alterações nos campos ["email", "password", "confirmPassword"].forEach(key => { const fn = form[key]; if (!fn) return; fn().addEventListener("input", toggleErrors); }); });
